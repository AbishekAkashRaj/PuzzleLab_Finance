  async function saveSimulation() {
    if (result === null) {
      alert("Please run the simulation first.");
      return;
    }

    const token = localStorage.getItem("token");

    if (!token) {
      alert("Please login first.");
      return;
    }

    try {
      const response = await fetch("http://127.0.0.1:5050/api/simulations", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          Authorization: `Bearer ${token}`,
        },
        body: JSON.stringify({
          name: "Equity Premium Simulation",
          riskAversion,
          disasterProbability,
          disasterSize,
          premium: result,
        }),
      });

      if (!response.ok) {
        throw new Error("Failed to save simulation");
      }

      alert("Simulation saved successfully.");
    } catch (error) {
      console.error(error);
      alert("Backend save failed.");
    }
  }
